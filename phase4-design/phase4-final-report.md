### 配置模板

```yaml
# 任务执行优化配置（最终版）
retry:
  strategy: progressive_timeout
  max_retries: 2
  timeouts: [180s, 240s, 300s]
  retry_on: [timeout, rate_limit, temporary_failure]

  enabled: true
  save_per_step: true
  retention_on_failure: 86400s  # 24h
  cleanup_on_success: 300s

idempotency:
  key_format: "{task_id}:{step_id}:{attempt}"
  storage: local_file

resource_limits:
  retry_budget_per_hour: 10
  circuit_breaker_threshold: 5
  recovery_timeout: 300s
```
